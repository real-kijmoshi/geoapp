from js import document, FileReader def handle_file_upload(event): file_input = document.getElementById("file") if file_input.files.length > 0: uploaded_file = file_input.files[0] file_name = uploaded_file.name output = document.getElementById("output") output.innerHTML = f"File uploaded: {file_name}" if uploaded_file.type == 'text/plain': reader = FileReader.new() reader.readAsText(uploaded_file) def onload(event): file_content = reader.result output.innerHTML += f"
File content: 
{file_content}
" reader.onload = to_js(onload) else: output.innerHTML += "
Only text files are supported for reading." else: output = document.getElementById("output") output.innerHTML = "No file selected." button = document.getElementById("calculateButton") button.addEventListener("click", handle_file_upload)
